// use_imports
extern uint64_t g_count;
extern FILE* stdout;

char* use_imports(char* a1, uint64_t a2) {
  if (a1) {
    uint64_t r_strlen = strlen(a1);
    g_count += r_strlen;
    puts(a1);
  }
  if (a2) {
    fputs(a2, stdout);
  }
  char* r_getenv = getenv("HOME");
  if (r_getenv) {
    g_count++;
    return g_count + 1;
  }
  return r_getenv;
}
